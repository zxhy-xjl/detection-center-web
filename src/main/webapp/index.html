    方法名称：RCU.Terminal.info() 读取终端信息
    输入参数：无
    输出结果：终端rcu版本和终端唯一编码信息（json对象经序列化后的字符串）

        9其他

	    terminal:终端唯一编码信息
	  },
	  error:{
	    code:1,
	    message:没找到RCU
	  }
	}

			        2无效身份证
			        9其他
		
    方法名称：RCU.ThermoPrinter.print() 打印小票

    输入参数:序列化后的json对象：

 {
   content: <经过base64编码过的图片数据> ,
   callback:回调函数名称
 }

    输出结果：无
    
    样例
//调用打印
var img = 'base64编码的图片';
var content = {content:img,callbackName:'callbackName'};
var cont = JSON.stringify(content);

RCU.ThermoPrinter.print(cont);

//获取打印结果
function callbackName(code, message){
   if (code == 0){
     console.log("打印成功");
   } else {
     console.log("打印失败", code + "(" + message + ")");
   }
}
    

    1.打印内容

	方法名称：RCU.A4Printer.printContent() 
	输入参数:json数据格式（与热敏打印机一致）
	{
	   content:,
	   callbackName:<回调函数名称>
	}
	输出结果：无

    2.打印文件

	方法名称：RCU.A4Printer.printFile() 
	输入参数:json数据格式
	{
	   filePath:<文件的全路径，比如E:\test.doc>,
	   callbackName:<回调函数名称>
	}
	输出结果：无

    3.打印远端url

	方法名称：RCU.A4Printer.printUrl() 
	输入参数:json数据格式
	{
	   fileUrl:<远端文件的地址，比如http://homepages.inf.ed.ac.uk/neilb/TestWordDoc.doc>,
	   callbackName:<回调函数名称>
	}
	输出结果：无

    回调函数

	// code:代码，message:消息
	function callbackName (code, message) {
	}
	
	code代码
	0:打印机等待（打印成功）
	1:用户输入数据错误
	2:文件未找到
	3:打印机离线
	4:打印机忙碌
	5:打印机缺纸
	99:其他错误
	
	    示例
	
	//打印内容
	var img = 'base64编码的图片';
	var content = {content:img,callbackName:'callbackName'};
	var cont = JSON.stringify(content);
	RCU.A4Printer.printContent(cont);
	
	//打印本地文件
	var file = 'E:\\test.doc';
	var content = {filePath:file,callbackName:'callbackName'};
	var cont = JSON.stringify(content);
	
	RCU.A4Printer.printFile(cont);
	
	//打印远端url
	var url = 'http://homepages.inf.ed.ac.uk/neilb/TestWordDoc.doc';
	var content = {fileUrl:url,callbackName:'callbackName'};
	var cont = JSON.stringify(content);
	RCU.A4Printer.printUrl(cont);
	
	//回调函数获取打印结果
	function callbackName(code, message){
	  if (code == 0){
	    console.log("打印成功");
	  } else {
	    console.log("打印失败", code + "(" + message + ")");
	  }
	}


    方法名称：RCU.HSI.open(callbackName) 打开高拍仪

	输入参数：回调函数名称
	输出结果：无
	返回结果：json数据格式
	{
	  success: ,
	  error: {
	    code: <错误代码>,
	    message: <错误消息>
	  } 
	}
	
	code代码
	1：高拍仪打开失败
	2：回调函数为空
	9：其他错误
	
	    回调函数
	
	// img: 经过base64编码的图像数据
	function callbackName (img) {
	}
	
	    示例
	
	var ret = RCU.HSI.open('HSINewImage');
	var result = JSON.parse(ret);
	if (result.success != true) {
	  alert('高拍仪打开错误'+ result.error.code + '(' + result.error.message + ')');
	}
	
	 //回调函数获取图片
	function callbackName(image){
	  img.src = 'data:image/jpeg;base64,' + image;
	}


    方法名称1：RCU.Camera.play(videoId, callback) 播放摄像头画面
        输入参数：videoId:video元素的id; callback:回调函数名称
        输出结果：视频在html上的video元素上播放
        播放的结果会通过callback返回给用户

    方法名称2：RCU.Camera.snap(video, canvas) 拍照
        输入参数：videoId：video元素的id；canvasId：canvas元素的Id
        输出结果：拍照的内容在canvas元素上显示

    方法名称3：RCU.Camera.stop() 停止
        输入参数：无
        输出结果：无（摄像头停止）

    回调函数：function callback (result) {}
        输入参数：result:播放摄像头画面的结果，一个序列化后的json对象，如下：

{
  success:,
  error:{
    code:,
    message:
  }
}
